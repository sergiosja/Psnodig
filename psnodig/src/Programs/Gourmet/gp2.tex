\proc{$InsertionSort(A)$}{
	\For{$i \gets 1$ \KwTo $\var{n} - 1$}{
		$\var{j} \gets \var{i}$ \;
		\While{$\var{j} > 0 \land \var{A}[\var{j} - 1] > \var{A}[\var{j}]$}{
			swap(\var{A}[\var{j} - 1], \var{A}[\var{j}]) \;
			$\var{j} \gets \var{j} - 1$ \;
		}
	}
