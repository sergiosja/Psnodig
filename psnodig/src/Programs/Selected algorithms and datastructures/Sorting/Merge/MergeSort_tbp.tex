\begin{algorithm}[H]
\proc{$\Merge(A1, A2, A)$}{
	$\texttt{i} \gets 0$ \;
	$\texttt{j} \gets 0$ \;
	\While{$i < \abs{A1} \land j < \abs{A2}$}{
		\uIf{$A1[i] < A2[j]$}{
			$A[i + j] \gets A1[i]$ \;
			$\texttt{i} \gets i + 1$ \;
		}
		\uElse{
			$A[i + j] \gets A2[j]$ \;
			$\texttt{j} \gets j + 1$ \;
		}
	}
	\While{$i < \abs{A1}$}{
		$A[i + j] \gets A1[i]$ \;
		$\texttt{i} \gets i + 1$ \;
	}
	\While{$j < \abs{A2}$}{
		$A[i + j] \gets A2[j]$ \;
		$\texttt{j} \gets j + 1$ \;
	}
	\Return A \;
}
\caption{Merge}
\end{algorithm}
