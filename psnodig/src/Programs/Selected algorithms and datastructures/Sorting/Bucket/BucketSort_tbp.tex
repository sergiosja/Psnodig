\begin{algorithm}[H]
\proc{$\BucketSort(A)$}{
	
	\text{B $\gets$ Array of N empty lists} \;
	
	\For{$i \gets 0$ \KwTo $n - 1$}{
		\text{associate k with A[i]} \;
		append A[i] to B[k] \;
	}
	\text{} \;
	
	\For{$i \gets 0$ \KwTo $N' - 1$}{
		$B[i] \gets \InsertionSort{B[i]}$ \;
	}
	\text{} \;
	$\texttt{j} \gets 0$ \;
	\For{$k \gets 0$ \KwTo $N - 1$}{
		\For{$x \in B[k]$}{
			$A[j] \gets x$ \;
			$\texttt{j} \gets j + 1$ \;
		}
	}
	\Return A \;
}
\caption{BucketSort}
\end{algorithm}
