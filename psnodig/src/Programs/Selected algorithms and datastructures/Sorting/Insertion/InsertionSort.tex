\begin{algorithm}[H]
\KwIn{Input}
\KwOut{Output}
\proc{$\InsertionSort(\A)$}{
	
	\For{$i \gets 1$ \KwTo $n - 1$}{
		$\var{j} \gets i$ \;
		\While{$j > 0 \land \A{j - 1} > \A{j}$}{
			\text{swap(A[j], A[j-1])} \;
			$\var{j} \gets j - 1$ \;
		}
	}
}\caption{InsertionSort}
\end{algorithm}
