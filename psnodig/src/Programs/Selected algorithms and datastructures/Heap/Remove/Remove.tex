\begin{algorithm}[H]
\KwIn{Input}
\KwOut{Output}
\proc{$\Remove(\A)$}{
	
	$\var{x} \gets \A[0]$ \;
	$\A[0] \gets \A[n - 1]$ \;
	$\var{i} \gets 0$ \;
	\While{$2 \cdot i + 2 < n - 1$}{
		\text{j $\gets$ if A[2$\cdot$i+1] $<=$ A[2$\cdot$i+2] then 2$\cdot$i+1 else 2$\cdot$i+2} \;
		\uIf{$\A[j] \leq \A[i]$}{
			\text{swap A[j] and A[i]} \;
			$\var{i} \gets j$ \;
			\KwContinue \;
		}
		\KwBreak \;
	}
	\uIf{$2 \cdot i + 1 < n - 1 \land \A[2 \cdot i + 1] \leq \A[i]$}{
		\text{swap A[i] and A[2$\cdot$i + 1]} \;
	}
	\Return $x$ \;
}\caption{Remove}
\end{algorithm}
