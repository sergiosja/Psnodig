\begin{algorithm}[H]
\KwIn{Input}
\KwOut{Output}
\proc{$\Height(node, a, b)$}{
	h $\gets$ $-1$ \;
	\uIf{node $=$ \KwNil}{
		\Return h \;
	}
	\For{$v' \in node_{children} $}{
		h $\gets$ max(\h, Height(v')) \;
	}
	\Return $1$ $+$ h \;
}\caption{Height}
\end{algorithm}

\begin{algorithm}[H]
\KwIn{Input}
\KwOut{Output}
\proc{$\f()$}{
	t $\gets$ \Tree($5$, \KwNil, [\Tree($5$, \KwNil, [\Tree($5$, \KwNil, [])])]) \;
	print(Height(t)) \;
	\Return $1$ \;
}\caption{f}
\end{algorithm}
