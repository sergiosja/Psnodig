\begin{algorithm}[H]
\KwIn{Input}
\KwOut{Output}
\proc{$\BucketSort(\A, N)$}{
	\text{B := Array of N empty lists} \;
	
	\For{$i \gets 0$ \KwTo $n - 1$}{
		\text{associate k with A[i]} \;
		append \A{i} to \B{k} \;
	}
	$\var{j} \gets 0$ \;
	\For{$k \gets 0$ \KwTo $N - 1$}{
		\For{$x \in \B{k}$}{
			$\A{j}\gets x$ \;
			$\var{j} \gets j + 1$ \;
		}
	}
	\Return $A$ \;
}\caption{BucketSort}
\end{algorithm}
