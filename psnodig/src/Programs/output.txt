\documentclass{standalone}
\usepackage[utf8]{inputenc}
\usepackage{amsmath,commath} 
\usepackage[linesnumbered, ruled]{algorithm2e}
\SetKwProg{proc}{Procedure}{}{}
\SetKw{KwFalse}{false}
\SetKw{KwTrue}{true}
\SetKw{KwPrint}{print}
\newcommand{\var}{\texttt}
\DontPrintSemicolon
\begin{document}

\begin{algorithm}[H]
\KwIn{Input}
\KwOut{Output}
\proc{$bs(A, x)$}{
	$\var{ans} \gets bs'(\var{A}, \var{x}, 0, len(\var{A}))$ \;
	\KwPrint $\var{ans}$ \;
}\caption{Hva heter algoritmen?}
\end{algorithm}

\begin{algorithm}[H]
\KwIn{Input}
\KwOut{Output}
\proc{$bs'(A, x, low, high)$}{
	$\var{high} \gets \var{high} - 1$ \;
	\uIf{$\var{low} \geq \var{high}$}{
		\Return $\KwFalse$
	}
	$\var{middle} \gets \var{low} + \var{high} / 2$ \;
	\uIf{$\var{A}[\var{middle}] = \var{x}$}{
		\Return $\KwTrue$
	}
	\uIf{$\var{A}[\var{middle}] < \var{x}$}{
		$\var{newlow} \gets \var{low} + 1$ \;
		\Return $bs'(\var{A}, \var{x}, \var{newlow}, \var{high})$
	}
	\uIf{$\var{A}[\var{middle}] > \var{x}$}{
		$\var{newhigh} \gets \var{high} - 1$ \;
		\Return $bs'(\var{A}, \var{x}, \var{low}, \var{newhigh})$
	}
}\caption{Hva heter algoritmen?}
\end{algorithm}

\end{document}
