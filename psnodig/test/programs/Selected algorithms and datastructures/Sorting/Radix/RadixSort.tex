\begin{algorithm}[H]
\KwIn{Input}
\KwOut{Output}
\proc{$\RadixSort(\A)$}{
	\text{d := number of digits in the greatest number} \;
	\For{$i \gets d$ \KwTo $0$}{
		$\var{A} \gets \BucketSort(A, d)$ \;
	}
	\Return $A$ \;
}\caption{RadixSort}
\end{algorithm}
