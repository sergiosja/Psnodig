\begin{algorithm}[H]
\KwIn{Input}
\KwOut{Output}
\proc{$\SelectionSort(\A)$}{
	
	\For{$i \gets 0$ \KwTo $n - 1$}{
		$\var{k} \gets i$ \;
		\For{$j \gets j + 1$ \KwTo $n - 1$}{
			\uIf{$\A{j} < \A{k}$}{
				$\var{k} \gets j$ \;
			}
		}
		\uIf{$i \neq k$}{
			\text{swap(A[i], A[k])} \;
		}
	}
}\caption{SelectionSort}
\end{algorithm}
