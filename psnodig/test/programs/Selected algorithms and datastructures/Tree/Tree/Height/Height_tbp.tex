\begin{algorithm}[H]
\proc{$\Height(node)$}{
	$\texttt{h} \gets -1$ \;
	\uIf{$node = \KwNil$}{
		\Return h \;
	}
	\For{$v' \in node_{children} $}{
		$\texttt{h} \gets \max{h, \Height{v'}}$ \;
	}
	\Return 1 + h \;
}
\caption{Height}
\end{algorithm}
