\begin{algorithm}[H]
\KwIn{Input}
\KwOut{Output}
\proc{$\Search(\A, x)$}{
	
	$\var{i} \gets 0$ \;
	\While{$i \neq n$}{
		\uIf{$\A{i} = x$}{
			\Return $\KwTrue$ \;
		}
		$\var{i} \gets i + 1$ \;
	}
	\Return $\KwFalse$ \;
}\caption{Search}
