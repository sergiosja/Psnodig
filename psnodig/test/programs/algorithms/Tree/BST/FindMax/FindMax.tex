\begin{algorithm}[H]
\KwIn{Input}
\KwOut{Output}
\proc{$\FindMax(root)$}{
	\uIf{$root = \KwNil$}{
		\Return $\KwNil$ \;
	}
	\uIf{$root_{right}  = \KwNil$}{
		\Return $root$ \;
	}
	\Return $\FindMax{root_{right} }$ \;
}\caption{FindMax}
\end{algorithm}
