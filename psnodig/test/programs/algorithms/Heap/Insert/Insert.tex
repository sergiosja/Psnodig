\begin{algorithm}[H]
\KwIn{Input}
\KwOut{Output}
\proc{$\Insert(\A, x)$}{
	\text{A[n] pil x} \;
	
	$\var{i} \gets n$ \;
	\While{$0 < i \land \A[i] < \A[\lfloor \frac{i - 1}{2}\rfloor]$}{
		\text{swap A[i] and A[floor((i-1)/2)]} \;
		$\var{i} \gets \lfloor \frac{i - 1}{2}\rfloor$ \;
	}
	\Return $A$ \;
}\caption{Insert}
\end{algorithm}

\begin{algorithm}[H]
\KwIn{Input}
\KwOut{Output}
\proc{$\f(\A)$}{
	\print(\Insert(\A, 2)) \;
	\Return $1$ \;
}\caption{f}
\end{algorithm}
